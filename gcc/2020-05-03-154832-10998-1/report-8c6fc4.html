574 fprintf(stderr, " IV=%s\n", showdata(aes_iv,0));
575*/
576 buf_aescrypt(encrypted, aes_tte_key,
577 aes_iv, ENCRYPT1);
578 buf_aescrypt(body, aes_body_key, aes_iv,
579 ENCRYPT1);
580 buf_aescrypt(other, aes_header_key,
581 aes_iv, ENCRYPT1);
582
583 /* Only 2*512 headers covered by digest so no remailer can tell where it is in the chain. */
584 if (!hop) {
585 /* Hop 0 (final) should not have a valid HMAC of the following heaer.
586 * If it did that would tell the exit remailer whether the chain was
587 * maximum length (e.g. 10 hops with large keys).
588 */
589 buf_reset(temp);
590 buf_setrnd(temp, 32);
591 } else {
592 buf_reset(antitag);
593 buf_append(antitag, other->data,
594 2 * 512);
595 buf_reset(temp);
596 hmac_sha256(antitag, hkey,
597 temp);
598 }
599 /* antitag */ buf_cat(key, temp);
600 buf_reset(temp);
601 hmac_sha256(body, hkey, temp);
602 /* body */ buf_cat(key, temp);
603
604 buf_reset(temp);
605 hmac_sha256(encrypted, hkey, temp);
606 /* encrypted-328-block */ buf_cat(key,
607 temp);
608
609 /* AES pre key */ buf_cat(key,
610 aes_pre_key);
611
612 err = pk_encrypt(
613 key,
614 pubkey); /* ... and encrypt the session key etc */
615 whiner += 2;
616 }
617
618 buf_free(pubkey);
619 if (err == -1 ||
620 (key->length != 128 && key->length != 256 &&
621 key->length != 384 &&
622 key->length != 512)) {
623 clienterr(feedback,
624 "Encryption failed!");
625 err = -1;
626 goto end;
627 }
628 if (whiner > 20) {
629 clienterr(
630 feedback,
631 "Chain too long. Limit is 20 of 1k keys, or 10 of large keys, or somewhere between.");
632 err = -1;
633 goto end;
634 }
635
636 /* now build the new header */
637 buf_clear(header);
638 buf_append(header,
639 remailer[thischain[hop]].keyid, 16);
640 /* one byte to show RSA length */
641 switch (key->length) {
642 case 128:
643 /* Legacy 1024-bit RSA means 128 bytes. */
644 buf_appendc(header, 128);
645 break;
646 case 256:
647 buf_appendc(header, 2); /* 2048 */
648 break;
649 case 384:
650 buf_appendc(header, 3); /* 3072 */
651 break;
652 case 512:
653 buf_appendc(header, 4); /* 4096 */
654 break;
655 default:
656 clienterr(
657 feedback,
658 "RSA key size not acceptable!");
659 err = -1;
660 goto end;
661 break;
662 }
663 buf_cat(header, key);
664 buf_cat(header, iv);
665 buf_cat(header, encrypted);
666 buf_pad(header,
667 key->length == 128 ?
668 512 :
669 1024); /* 512 bytes if RSA is 1024 bits */
670 if (128 == key->length)
671 buf_cat(header, other);
672 else
673 buf_append(header, other->data,
674 18 * 512);
675 break;
676 default:
677 err = -1;
678 goto end;
679 }
680 }
681
682 /* build the message */
683 buf_sets(out, remailer[thischain[chainlen - 1]].addr);
684 buf_nl(out);
685 buf_cat(out, header);
686 buf_cat(out, body);
687 assert(header->length == 10240 && body->length == 10240)((void) (0));
688 mix_pool(out, INTERMEDIATE0, -1);
689
690 if (feedback) {
691 for (hop = chainlen - 1; hop >= 0; hop--) {
692 buf_appends(feedback,
693 remailer[thischain[hop]].name);
694 if (hop > 0)
695 buf_appendc(feedback, ',');
696 }
697 buf_nl(feedback);
698 }
699 }
700
701end:
702 buf_free(aes_body_key);
703 buf_free(aes_header_key);
704 buf_free(aes_iv);
705 buf_free(aes_pre_key);
706 buf_free(aes_tte_key);
707 buf_free(antitag);
708 buf_free(body);
709 buf_free(encrypted);
710 buf_free(header);
711 buf_free(hkey);
712 buf_free(iv);
713 buf_free(ivarray);
714 buf_free(key);
715 buf_free(other);
716 buf_free(out);
717 buf_free(pid);
718 buf_free(temp);
719
720 return (err);
721}

723int redirect_message(BUFFER *sendmsg, char *chainstr, int numcopies,
724 BUFFER *feedback)
725{
726 BUFFER *field;
727 BUFFER *content;
728 BUFFER *line;
729 char recipient[80] = "";
730 int num = 0;
731 int err = 0;
732 int c;
733 int hop;
734
735 REMAILER remailer[MAXREM100];
736 int chain[20];
737 int thischain[20];
738 int chainlen;
739 int badchains[MAXREM100][MAXREM100];
740 int maxrem;
741 int tempchain[20];
742 int tempchainlen;
743 int israndom;
744
745 field = buf_new();
746 content = buf_new();
747 line = buf_new();
748
749 if (numcopies == 0)
750 numcopies = NUMCOPIES;
751 if (numcopies > 10)
752 numcopies = 10;
753
754 /* Find the recipient */
755 while (buf_getheader(sendmsg, field, content) == 0)
756 if (bufieq(field, "to")) {
757 strncpy0(recipient, content->string, sizeof(recipient));
758 num++;
759 };
760 if (num != 1) {
761 clienterr(feedback, "Did not find exactly one To: address!");
762 err = -1;
763 goto end;
764 };
765
766 /* Dearmor the message */
767 err = mix_dearmor(sendmsg, sendmsg);
768 if (err == -1)
769 goto end;
770 assert(sendmsg->length == 20480)((void) (0));
771
772 /* Check the chain */
773 maxrem = mix2_rlist(remailer, badchains);
774 if (maxrem < 1) {
775 clienterr(feedback, "No remailer list!");
776 err = -1;
777 goto end;
778 }
779 chainlen = chain_select(chain, chainstr, maxrem, remailer, 0, line);
780 if (chainlen < 1) {
781 if (line->length)
782 clienterr(feedback, line->string);
783 else
784 clienterr(feedback, "Invalid remailer chain!");
785 err = -1;
786 goto end;
787 } else if (chainlen >= 20) {
788 clienterr(
789 feedback,
790 "A chainlength of 20 will certainly destroy the message!");
791 err = -1;
792 goto end;
793 };
794
795 for (c = 0; c < numcopies; c++) {
796 /* if our recipient is a remailer we want to make sure we're not using a known broken chain.
797 * therefore we need to pick the final remailer with care */
798 for (hop = 0; hop < chainlen; hop++)
799 thischain[hop] = chain[hop];
800 if (thischain[0] == 0) {
801 /* Find out, if recipient is a remailer */
802 tempchainlen = chain_select(tempchain, recipient,
803 maxrem, remailer, 0, line);
804 if (tempchainlen < 1 && line->length == 0) {
805 /* recipient is apparently not a remailer we know about */
806 ;
807 } else {
808 /* Build a new chain, based on the one we already selected but
809 * with the recipient as the final hop.
810 * This is so that chain_rand properly selects nodes based on
811 * broken chains and DISTANCE */
812 assert(chainlen < 20)((void) (0));
813 for (hop = 0; hop < chainlen; hop++)
814 thischain[hop + 1] = thischain[hop];
815 thischain[0] = tempchain[0];
816
817 israndom = chain_rand(remailer, badchains,
818 maxrem, thischain,
819 chainlen + 1, 0, 0);
820 if (israndom == -1) {
821 err = -1;
822 clienterr(feedback,
823 "No reliable remailers!");
824 goto end;
825 }
826
827 /* Remove the added recipient hop */
828 for (hop = 0; hop < chainlen; hop++)
829 thischain[hop] = thischain[hop + 1];
830 };
831 };
832
833 /* queue the packet */
834 if (send_packet(1, sendmsg, thischain, chainlen, -1, -1, NULL((void*)0),
835 remailer, badchains, maxrem, recipient, 0,
836 feedback) == -1)
837 err = -1;
838 };
839
840end:
841 buf_free(field);
842 buf_free(content);
843 buf_free(line);
844 return (err);
845}

847int mix2_encrypt(int type, BUFFER *message, char *chainstr, int numcopies,
848 int ignore_constraints_if_necessary, BUFFER *feedback)
849{
850 /* returns 0 on success, -1 on error. feedback contains the selected
851 remailer chain or an error message
852
853 ignore_constraints_if_necessary .. to be used when randhopping messages.
854 if a chain can not be constructed otherwhise,
855 maxlat, minlat, and minrel are ignored.
856 */
857
858 REMAILER remailer[MAXREM100];
859 int badchains[MAXREM100][MAXREM100];
860 int maxrem;
861 BUFFER *line, *field, *content, *header, *msgdest, *msgheader, *body,
862 *temp, *mid;
863 byte numdest = 0, numhdr = 0;
864 char hdrline[LINELEN128];
865 BUFFER *packet;
866 int chain[20];
867 int chainlen;
868 size_t i;
869 int err = 0;
870
871 mix_init(NULL((void*)0));
872 packet = buf_new();
873 line = buf_new();
874 field = buf_new();
875 content = buf_new();
876 msgheader = buf_new();
877 msgdest = buf_new();
878 body = buf_new();
879 temp = buf_new();
880 mid = buf_new();
881 header = buf_new();
882 if (feedback)
883 buf_reset(feedback);
884
885 if (numcopies == 0)
886 numcopies = NUMCOPIES;
887 if (numcopies > 10)
888 numcopies = 10;
889
890 maxrem = mix2_rlist(remailer, badchains);
891 if (maxrem < 1) {
892 clienterr(feedback, "No remailer list!");
893 err = -1;
894 goto end;
895 }
896 chainlen = chain_select(chain, chainstr, maxrem, remailer, 0, line);
897 if (chainlen < 1) {
898 if (line->length)
899 clienterr(feedback, line->string);
900 else
901 clienterr(feedback, "Invalid remailer chain!");
902 err = -1;
903 goto end;
904 }
905 if (chain[0] == 0)
906 chain[0] = chain_randfinal(type, remailer, badchains, maxrem, 0,
907 chain, chainlen,
908 ignore_constraints_if_necessary);
909
910 if (chain[0] == -1) {
911 clienterr(feedback, "No reliable remailers!");
912 err = -1;
913 goto end;
914 }
915 switch (remailer[chain[0]].version) {
916 case 2:
917 if (type == MSG_NULL0) {
918 memset(hdrline, 0, 80);
919 strcpy(hdrline, "null:");
920 buf_appends(msgdest, hdrline);
921 numdest++;
922 } else
923 while (buf_getheader(message, field, content) == 0) {
924 if (bufieq(field, "to")) {
925 memset(hdrline, 0, 80);
926 strncpy0(hdrline, content->string, 80);
927 buf_appends(msgdest, hdrline);
928 numdest++;
929 } else if (type == MSG_POST2 &&
930 bufieq(field, "newsgroups")) {
931 memset(hdrline, 0, 80);
932 strcpy(hdrline, "post: ");
933 strcatn(hdrline, content->string, 80);
934 buf_appends(msgdest, hdrline);
935 numdest++;
936 } else {
937 buf_clear(header);
938 buf_appendheader(header, field,
939 content);
940 hdr_encode(header, 80);
941 while (buf_getline(header, line) == 0) {
942 /* paste in encoded header entry */
943 memset(hdrline, 0, 80);
944 strncpy0(hdrline, line->string,
945 80);
946 buf_appends(msgheader, hdrline);
947 numhdr++;
948 }
949 }
950 }
951 buf_appendc(body, numdest);
952 buf_cat(body, msgdest);
953 buf_appendc(body, numhdr);
954 buf_cat(body, msgheader);
955
956 if (type != MSG_NULL0) {
957 buf_rest(temp, message);
958 if (temp->length > 10236 &&
959 remailer[chain[0]].flags.compress)
960 buf_compress(temp);
961 buf_cat(body, temp);
962 buf_reset(temp);
963 }
964 buf_setrnd(mid, 16); /* message ID */
965 for (i = 0; i <= body->length / 10236; i++) {
966 long length;
967
968 length = body->length - i * 10236;
969 if (length > 10236)
970 length = 10236;
971 buf_clear(packet);
972 buf_appendl_lo(packet, length);
973 buf_append(packet, body->data + i * 10236, length);
974 buf_pad(packet, 10240);
975 if (send_packet(numcopies, packet, chain, chainlen,
976 i + 1, body->length / 10236 + 1, mid,
977 remailer, badchains, maxrem, NULL((void*)0),
978 ignore_constraints_if_necessary,
979 feedback) == -1)
980 err = -1;
981 }
982 break;
983 case 3:
984 NOT_IMPLEMENTED{ printf("Function not implemented.\n"); return -1; };
985 break;
986 default:
987 fprintf(stderrstderr, "%d\n", chain[0]);
988 clienterr(feedback, "Unknown remailer version!");
989 err = -1;
990 }
991
992end:
993 buf_free(packet);
994 buf_free(line);
995 buf_free(field);
996 buf_free(content);
997 buf_free(header);
998 buf_free(msgheader);
999 buf_free(msgdest);
1000 buf_free(body);
1001 buf_free(temp);
1002 buf_free(mid);
1003 return (err);
