352/* float chain_reliablity(char *chain, int chaintype,
353 char *reliability_string);
354 *
355 * Compute reliablity of a chain.
356 *
357 * We get the reliablity of the chain by multiplying the reliablity of
358 * every remailer in the chain. The return value is the reliablity of
359 * the chain, or a negative number if the reliablity can not be
360 * calculated. There are two reasons why may not be able to calculated
361 * the reliablity: A remailer in the chain is selected randomly, or we
362 * don't have statistics about one of the remailers in the chain.
363 * remailer_type indicates the remailer type:
364 * 0 = Mixmaster, 1 = Cypherpunk
365 *
366 * If reliability_string is non-NULL, the reliability is also returned
367 * as a string in this variable. The size of the string must be at
368 * least 9 characters!
369 *
370 * This function has been added by Gerd Beuster. (gb@uni-koblenz.de)
371 *--------------------------------------------------------------------*/

373float chain_reliability(char *chain, int chaintype, char *reliability_string)
374{
375 float acc_reliability = 1; /* Accumulated reliablity */
376 char *name_start, *name_end; /* temporary pointers used
377 in string scanning */
378 char remailer_name[20]; /* The length of the array is taken from mix3.h. */
379 int error = 0;
380 int maxrem;
381 int i;
382 int previous = -1;
383 REMAILER remailer[MAXREM100];
384 int badchains[MAXREM100][MAXREM100];
385
386 /* chaintype 0=mix 1=ek 2=newnym */
387 assert((chaintype == 0) || (chaintype == 1))((void) (0));
388 maxrem = (chaintype == 0 ? mix2_rlist(remailer, badchains) :
389 t1_rlist(remailer, badchains));
390
391 /* Dissect chain */
392 name_start = chain;
393 name_end = chain;
394 while (*name_end != '\0') { /* While string not scanned completely */
395 do /* Get next remailer */
396 name_end += sizeof(char);
397 while ((*name_end != ',') && (*name_end != '\0'));
398 strncpy0(remailer_name, name_start,
399 (name_end - name_start) / sizeof(char) +
400 1 * sizeof(char));
401 remailer_name[name_end - name_start] = '\0';
402 /* Lookup reliablity for remailer remailer_name */
403 for (i = 0; (i < maxrem) &&
404 (strcmp(remailer[i].name, remailer_name) != 0);
405 i++)
406 ;
407 if (!strcmp(remailer[i].name, remailer_name)) { /* Found it! */
408 acc_reliability *=
409 ((float)remailer[i].info[chaintype].reliability) /
410 10000;
411 if (previous != -1) {
412 if (badchains[previous][i] || badchains[0][i])
413 acc_reliability = 0;
414 }
415 previous = i;
416 } else
417 error = 1; /* Did not find this remailer. We can't calculate
418 the reliablity for the whole chain. */
419 name_start = name_end + sizeof(char);
420 }
421
422 if (error || (name_start == name_end))
423 acc_reliability = -1;
424
425 /* Convert reliability into string, if appropriate */
426 if (reliability_string) {
427 if (acc_reliability < 0)
428 sprintf(reliability_string, " n/a ");

431 acc_reliability * 100);

434 }
